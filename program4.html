Program 4: WRITE A C PROGRAM TO DISPLAY THE FOLLOWING BY READING THE NUMBER OF ROWS AS INPUT, 1 1 2 1 1 2 3 2 1 1 2 3 4 3 2 1

Description: This program demonstrates a basic implementation of an array traversal.

Code:

 #include<stdio.h>
#include<stdlib.h>

void main()
{
int i,j,n;
clrscr( );
printf("Input number of rows : ");
scanf("%d",&n);
for(i=0; i<=n;i++)
{
for(j=1;j<=n-i;j++)
{
printf(" ");
}
for(j=1;j<=i;j++)
{
printf("%d",j);
}
for(j=i-1;j>=1;j--)
{
printf("%d",j);
}
printf("\n");
}
getch();
}
