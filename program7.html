Program 7: SINE FUNCTION USING TAYLOR SERIES

Description: This program calculates sine using the Taylor series.

Code:

#include <stdio.h>
#include <math.h>

int main() {
    int i, degree;
    float x, sum = 0, term, n;

    printf("Enter the value of degree: ");
    scanf("%d", °ree);
    printf("Enter the limit up to which the Taylor's series is to be calculated: ");
    scanf("%f", &n);
    
    x = degree * (3.1416 / 180);
    term = x;
    sum = term;

    for (i = 3; i <= n; i += 2) {
        term = -term * x * x / ((i - 1) * i);
        sum += term;
    }

    printf("The sine of %d is %.3f\n", degree, sum);
    printf("The sine function of %d using library function is %.3f\n", degree, sin(x));
    
    return 0;
}
