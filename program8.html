Program 8: SORT THE GIVEN SET OF N NUMBERS USING BUBBLE SORT.

Description: This program demonstrates a basic implementation of Bubble Sort.

Code:

#include <stdio.h>

int main()
{    
    int n, i, j, a[10], temp;
    
    printf("Enter the number of elements:\n");
    scanf("%d", &n);
    
    printf("Enter the array elements:\n");
    for(i = 0; i < n; i++)
        scanf("%d", &a[i]); 
    
    printf("The original elements are:\n");
    for(i = 0; i < n; i++)
        printf("%d ", a[i]);

    for(i = 0; i < n-1; i++) // Number of passes
    {
        for(j = 0; j < n-i-1; j++) // Comparisons
        {
            if(a[j] > a[j+1])
            {
                temp = a[j];   
                a[j] = a[j+1];    
                a[j+1] = temp;
            }
        }
    }

    printf("\nThe sorted elements are:\n"); 
    for(i = 0; i < n; i++)
        printf("%d\t", a[i]); 
    
    return 0;
}

Output:

5 4 3 2 1
The sorted elements are:
