Program 6: IMPLEMENT MATRIX MULTIPLICATION

Description: This program implements matrix multiplication and ensures that the rules of multiplication are validated.

Code:

#include <stdio.h>

#define MAX 10

int main() {
    int a[MAX][MAX], b[MAX][MAX], c[MAX][MAX];
    int m, n, p, q, i, j, k;

    printf("Enter the size of first matrix (rows cols): ");
    scanf("%d %d", &m, &n);

    printf("Enter the size of second matrix (rows cols): ");
    scanf("%d %d", &p, &q);

    if (n != p) {
        printf("Matrix multiplication is not possible! Number of columns of first matrix must equal number of rows of second matrix.\n");
        return 1;
    }

    printf("Enter the elements of first matrix (%dx%d):\n", m, n);
    for (i = 0; i < m; i++) {
        for (j = 0; j < n; j++) {
            scanf("%d", &a[i][j]);
        }
    }

    printf("Enter the elements of second matrix (%dx%d):\n", p, q);
    for (i = 0; i < p; i++) {
        for (j = 0; j < q; j++) {
            scanf("%d", &b[i][j]);
        }
    }

    for (i = 0; i < m; i++) {
        for (j = 0; j < q; j++) {
            c[i][j] = 0;
        }
    }

    for (i = 0; i < m; i++) {
        for (j = 0; j < q; j++) {
            for (k = 0; k < n; k++) {
                c[i][j] += a[i][k] * b[k][j];
            }
        }
    }

    printf("\nThe product of the two matrices is:\n");
    for (i = 0; i < m; i++) {
        for (j = 0; j < q; j++) {
            printf("%4d ", c[i][j]);
        }
        printf("\n");
    }

    return 0;
}
