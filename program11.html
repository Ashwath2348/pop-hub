Program 11: Compute Sum, Mean, and Standard Deviation Using Pointers

Description: This program uses pointers to compute the sum, mean, variance, and standard deviation of a set of numbers.

Program Code:

#include <stdio.h>
#include <math.h>

int main() {
    int n, i;
    float x[20], sum = 0, mean, variance = 0, deviation;

    printf("Enter the value of n: ");
    scanf("%d", &n);

    printf("Enter %d real values:\n", n);
    for (i = 0; i < n; i++) {
        scanf("%f", (x + i));
    }

    // Calculate sum
    for (i = 0; i < n; i++) {
        sum += *(x + i);
    }

    mean = sum / n;

    // Calculate variance
    for (i = 0; i < n; i++) {
        variance += (*(x + i) - mean) * (*(x + i) - mean);
    }
    variance /= n;

    deviation = sqrt(variance);

    // Display results
    printf("\nSum = %f\n", sum);
    printf("Mean (Average) = %f\n", mean);
    printf("Variance = %f\n", variance);
    printf("Standard Deviation = %f\n", deviation);

    return 0;
}
