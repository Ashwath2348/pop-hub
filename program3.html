Program 3: Electricity Bill

Description: This program demonstrates a basic implementation of an array traversal.

Code:

#include<stdio.h>
#include<stdlib.h>
void main()
{
char name[10];
float unit, amt;
clrscr();
printf("Enter your name and unit Consumed:");
scanf("%s %f",name,&unit);
if(unit<=200)
amt=unit*0.80+100;
else if((unit>200)&&(unit<=300))
amt=200*0.80+((unit-200)*0.90)+100;
else
amt=200*0.80+100*0.90+((unit-300)*1)+100;
if(amt>400)
amt=1.15*amt;
printf("Name: %s\n Unit=%f \n charge=%f ",name,unit,amt);
getch();
}
