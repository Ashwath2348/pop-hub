Program 10: STUDENT RECORDS

Description: This program manages student records, calculates total marks, and determines students above and below average.

Code:

#include <stdio.h>

struct student {
    char name[20];
    float m1, m2, m3, total;
};

int main() {
    int i, j, k, n, aboveavg[10], belowavg[10];
    float N, T = 0, AVG = 0;
    struct student s[10];

    printf("Enter number of students:\n");
    scanf("%d", &n);
    N = (float)n;

    printf("Enter the details of students:\n");
    for (i = 0; i < n; i++) {
        printf("Enter the details of student %d:\n", i + 1);
        printf("Enter name: ");
        scanf("%s", s[i].name);
        printf("Enter marks (m1, m2, m3): ");
        scanf("%f%f%f", &s[i].m1, &s[i].m2, &s[i].m3);
        s[i].total = s[i].m1 + s[i].m2 + s[i].m3;
        T += s[i].total;
    }
    AVG = T / N;

    printf("\nDetails of Students:\n");
    printf("Name\t m1\t m2\t m3\t total\n");
    for (i = 0; i < n; i++) {
        printf("%s\t%.2f\t%.2f\t%.2f\t%.2f\n", s[i].name, s[i].m1, s[i].m2, s[i].m3, s[i].total);
    }
    printf("\nAVG = %.2f\n", AVG);

    j = k = 0;
    for (i = 0; i < n; i++) {
        if (s[i].total > AVG) {
            aboveavg[j++] = i;
        } else {
            belowavg[k++] = i;
        }
    }

    printf("\nStudents scoring above avg:\n");
    for (i = 0; i < j; i++) {
        printf("%s %.2f\n", s[aboveavg[i]].name, s[aboveavg[i]].total);
    }
    
    printf("\nStudents scoring below avg:\n");
    for (i = 0; i < k; i++) {
        printf("%s %.2f\n", s[belowavg[i]].name, s[belowavg[i]].total);
    }
    return 0;
}
