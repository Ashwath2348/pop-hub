Program 10: Student Marks - Compute Average and List Above/Below Average

Description: This program reads, writes, and computes the average marks of students, then lists those who scored above and below the average.

Code:

#include <stdio.h>

struct student {
    char name[20];
    float m1, m2, m3, total;
};

void main() {
    int i, j = 0, k = 0, n, aboveavg[10], belowavg[10];
    float N, T = 0, AVG = 0;
    struct student s[10];

    printf("Enter number of students:\n");
    scanf("%d", &n);
    N = (float)n;

    printf("Enter the details of students:\n");
    for (i = 0; i < n; i++) {
        printf("Enter the details of student %d\n", i + 1);
        printf("Enter name: ");
        scanf("%s", s[i].name);
        printf("Enter marks for three subjects: ");
        scanf("%f%f%f", &s[i].m1, &s[i].m2, &s[i].m3);
        s[i].total = s[i].m1 + s[i].m2 + s[i].m3;
        T += s[i].total;
    }

    AVG = T / N;

    printf("\nDetails of Students:\n");
    printf("Name\tm1\tm2\tm3\ttotal\n");
    for (i = 0; i < n; i++) {
        printf("%s\t%.2f\t%.2f\t%.2f\t%.2f\n", s[i].name, s[i].m1, s[i].m2, s[i].m3, s[i].total);
    }
    printf("\nClass Average = %.2f\n", AVG);

    for (i = 0; i < n; i++) {
        if (s[i].total > AVG) {
            aboveavg[j] = i;
            j++;
        } else {
            belowavg[k] = i;
            k++;
        }
    }

    printf("\nStudents scoring above average:\n");
    for (i = 0; i < j; i++) {
        printf("%s %.2f\n", s[aboveavg[i]].name, s[aboveavg[i]].total);
    }

    printf("\nStudents scoring below average:\n");
    for (i = 0; i < k; i++) {
        printf("%s %.2f\n", s[belowavg[i]].name, s[belowavg[i]].total);
    }
}
