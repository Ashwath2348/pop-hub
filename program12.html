Program 12: Copy a Text File to Another

Description: This program reads a file and copies its contents to another file.

Program Code:

#include <stdio.h>
#include <stdlib.h> // For exit()
  
int main()
{
    FILE *fptr1, *fptr2;
    char filename[100], c;
  
    printf("Enter the filename to open for reading: \n");
    scanf("%s", filename);
  
    // Open file for reading
    fptr1 = fopen(filename, "r");
    if (fptr1 == NULL)
    {
        printf("Cannot open file %s \n", filename);
        exit(0);
    }
  
    printf("Enter the filename to open for writing: \n");
    scanf("%s", filename);
  
    // Open file for writing
    fptr2 = fopen(filename, "w");
    if (fptr2 == NULL)
    {
        printf("Cannot open file %s \n", filename);
        fclose(fptr1);
        exit(0);
    }
  
    // Read and copy contents
    c = fgetc(fptr1);
    while (c != EOF)
    {
        fputc(c, fptr2);
        c = fgetc(fptr1);
    }
  
    printf("\nContents copied to %s\n", filename);
  
    fclose(fptr1);
    fclose(fptr2);
    return 0;
}

Expected Output:

Enter the filename to open for reading: OLD.TXT
Enter the filename to open for writing: NEW.TXT

Contents copied to NEW.TXT

OLD.TXT
WELCOME TO C PROGRAMMING

NEW.TXT
WELCOME TO C PROGRAMMING
