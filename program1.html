Program 1: Arithmetic Operations

Description: This program demonstrates a basic implementation of arithmetic operations using switch-case.

Code:

#include <stdio.h>
#include <conio.h>

void main() {
    int num1, num2, result;
    float divresult;
    char op;
    
    clrscr();
    printf("Enter the operation to be done (+, -, *, /): \n");
    scanf(" %c", &op);
    printf("Enter the value of num1 and num2: \n");
    scanf("%d%d", &num1, &num2);

    switch(op) {
        case '+': 
            result = num1 + num2;
            printf("Addition = %d\n", result);
            break;
        case '-': 
            result = num1 - num2;
            printf("Subtraction = %d\n", result);
            break;
        case '*': 
            result = num1 * num2;
            printf("Multiplication = %d\n", result);
            break;
        case '/': 
            if (num2 != 0) {
                divresult = num1 / (float)num2;
                printf("Division = %.2f\n", divresult);
            } else {
                printf("Error: Division by zero is not allowed.\n");
            }
            break;
        default: 
            printf("Invalid Input, Try Again\n");
    }
    getch();
}
